metricData = ['Accuracy','F1']
def getAverageAccuracy( result, parameter = 'parameter' ):
    resultCopy = result.reset_index().copy();
    iterables = [resultCopy[parameter].unique(), metricData]
    multiIndex = pd.MultiIndex.from_product(iterables, names=['parameter','metric'])
    avgAccuracy= pd.DataFrame( index = multiIndex ,columns=['Train','Valid','Test'] )
    for idx, row in avgAccuracy.iterrows():
        avgAccuracy.loc[idx]['Train'] = py.mean(resultCopy[(resultCopy['metric'] == idx[1])&(resultCopy[parameter] == idx[0])]['Train'])
        avgAccuracy.loc[idx]['Valid'] = py.mean(resultCopy[(resultCopy['metric'] == idx[1])&(resultCopy[parameter] == idx[0])]['Valid'])
        avgAccuracy.loc[idx]['Test'] = py.mean(resultCopy[(resultCopy['metric'] == idx[1])&(resultCopy[parameter] == idx[0])]['Test'])
    return avgAccuracy
avgAccuracyValue = getAverageAccuracy(result)

def plotAvgAccuracy(avgResult):
    plt.close('all')
    result_para = avgResult.reset_index().set_index('metric')
    fig, ((ax1, ax2)) = plt.subplots(nrows=1, ncols=2)
    axpos = [ax1,ax2]
    plt.tight_layout()
    for ax, m in zip(axpos,metricData):
        t = result_para.loc[m]
        x = t['parameter']
        ax.set_title(m,fontsize=10)
        ax.semilogx( x, t['Train'].ravel(), 'r--', \
                 x, t['Valid'].ravel(), 'bs-', \
                 x, t['Test'].ravel(), 'g^-' ) 
    ax.get_xaxis().tick_bottom()
    ax.get_yaxis().tick_left()
    plt.show()
